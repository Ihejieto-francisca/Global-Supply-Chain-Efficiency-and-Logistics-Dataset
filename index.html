1. How do manufacturing costs correlate with defect rates?¶

import pandas as pd
import matplotlib.pyplot as plt

# Calculate correlation
correlation = df[['Manufacturing costs', 'Defect rates']].corr().iloc[0,1]

# Scatter plot
plt.figure(figsize=(10, 5))
plt.scatter(df['Manufacturing costs'], df['Defect rates'], alpha=0.6)
plt.title(f"Manufacturing Costs vs. Defect Rates (Correlation: {correlation:.2f})")
plt.xlabel("Manufacturing Costs")
plt.ylabel("Defect Rates")
plt.grid(True)
plt.show()

# Line chart (grouped by product type)
product_group = df.groupby('Product type').agg({'Manufacturing costs': 'mean', 'Defect rates': 'mean'}).reset_index()
plt.figure(figsize=(12, 6))
plt.plot(product_group['Product type'], product_group['Manufacturing costs'], marker='o', label='Manufacturing Costs')
plt.plot(product_group['Product type'], product_group['Defect rates'], marker='o', label='Defect Rates')
plt.title("Manufacturing Costs & Defect Rates by Product Type")
plt.xlabel("Product Type")
plt.ylabel("Value")
plt.legend()
plt.xticks(rotation=45)
plt.savefig("manufacturing cost vs rates.png")
plt.show()

7. Which suppliers have the shortest lead times and lowest defect rates?¶

In [18]:
supplier_stats = df.groupby('Supplier name').agg({'Lead time': 'mean', 'Defect rates': 'mean'}).reset_index()
supplier_stats['Score'] = supplier_stats['Lead time'] + supplier_stats['Defect rates']  # Lower score = better
top_suppliers = supplier_stats.nsmallest(5, 'Score')

plt.figure(figsize=(10, 6))
plt.barh(top_suppliers['Supplier name'], top_suppliers['Lead time'], label='Lead Time', color='lightgreen')
plt.barh(top_suppliers['Supplier name'], top_suppliers['Defect rates'], left=top_suppliers['Lead time'], label='Defect Rate', color='pink')
plt.title("Top Suppliers by Lead Time & Defect Rates")
plt.xlabel("Composite Score")
plt.ylabel("Supplier")
plt.legend()
plt.savefig("top suppliers.png")
plt.show()
